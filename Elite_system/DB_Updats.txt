CREATE proc [dbo].[Get_Paid]
@From date,
@To date,
@Medical_Type numeric(18,0),
@Paid int null
as
begin

if @Paid=1
SELECT * FROM [dbo].[V_Listing_Bonds] where ([Medical_Type]=@Medical_Type) and ([Bond_Date] between @From and @To) and( Debtor!=0 or Creditor!=0) order by Bond_Date
else if @Paid=2
SELECT * FROM [dbo].[V_Listing_Bonds] where ([Medical_Type]=@Medical_Type) and ([Bond_Date] between @From and @To) and( Debtor=0 and Creditor=0) order by Bond_Date
end
