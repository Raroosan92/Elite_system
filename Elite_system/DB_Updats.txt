create proc [dbo].[Check_Logs]
@Check_Number		 nvarchar(50) = null,
@Check_Type			int =null,
@Check_EmpName		nvarchar(80)=null,
@Check_MedicalName	nvarchar(90)=null,
@Check_Company	nvarchar(90)=null,
@Check_Barcode		nvarchar(50)=null,
@Check_Uname		nvarchar(50)=NULL,
@Check_CurrentDate	date,
@Check_Date			date,
@From date =NULL,
@To date =NULL,
@check char =NULL
as
begin

if @check='I'
INSERT INTO [dbo].[Log_Checks]
           ([Check_Number]
           ,[Check_Type]
           ,[Check_EmpName]
           ,[Check_MedicalName]
           ,[Check_Barcode]
           ,[Check_CurrentDate]
           ,[Check_Date]
		   ,[Check_Company]
		   ,[Check_Uname])
     VALUES
           (@Check_Number 
           ,@Check_Type 
           ,@Check_EmpName
           ,@Check_MedicalName
           ,@Check_Barcode
           ,@Check_CurrentDate
           ,@Check_Date
		   ,@Check_Company
		   ,@Check_Uname
		   )
			



else if @check='S'
SELECT *
  FROM [dbo].[Log_Checks]
